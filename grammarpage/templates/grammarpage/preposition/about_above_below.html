{% translate "1. About: Topic and Approximation" %}

{% translate "Subject + Verb + about + Topic/Quantity" %}

{% translate "Discussing Topics" %}

{% translate "When introducing the subject of discussion, thought, or content:" %}

  • {% translate "We had a conversation about climate change." %}
  • {% translate "The documentary is about marine life." %}
  • {% translate "What are you thinking about?" %}

{% translate "Approximate Quantities" %}

{% translate "When indicating rough estimates or approximations:" %}

  • {% translate "There were about fifty people at the event." %}
  • {% translate "It's about three miles to the nearest gas station." %}
  • {% translate "The project will take about two weeks to complete." %}

{% translate "Special Cases" %}

  • {% translate "Be about to (imminent action):" %}

    {% translate "I was about to leave when the phone rang." %}

  • {% translate "How about (suggestions):" %}

    {% translate "How about we go for a walk?" %}

{% translate "2. Above: Superior Position and Beyond" %}

{% translate "Subject + BE/Verb + above + Reference Point" %}

{% translate "Physical Position" %}

{% translate "When something is at a higher level than something else, often without direct contact:" %}

  • {% translate "The painting hangs above the fireplace." %}
  • {% translate "The helicopter hovered above the building." %}
  • {% translate "Temperatures remained above freezing all week." %}

{% translate "Abstract Concepts" %}

{% translate "When referring to higher status, rank, or quality:" %}

  • {% translate "Her work is above average." %}
  • {% translate "The general is above the colonel in rank." %}
  • {% translate "I consider that behavior above reproach." %}
