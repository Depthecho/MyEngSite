{% translate "These three prepositions are among the most frequently used in English, each serving distinct grammatical functions to express relationships of agency, purpose, and origin." %}

{% translate "1. The Preposition 'By'" %}

{% translate "Core Meanings and Uses" %}

{% translate "Function" %} {% translate "Explanation" %} {% translate "Examples" %}
{% translate "Agent in passive voice" %} {% translate "Indicates who performs the action" %} {% translate "The novel was written by Hemingway." %}
{% translate "Means/method" %} {% translate "Shows how something is done" %} {% translate "We traveled by plane." %}
{% translate "Time deadline" %} {% translate "Indicates a time before which something must happen" %} {% translate "Submit your work by Friday." %}
{% translate "Proximity" %} {% translate "Shows physical closeness" %} {% translate "She sat by the window." %}

{% translate "Special Constructions with 'By'" %}

  • {% translate "By + reflexive pronoun: To mean 'alone'" %}

    {% translate "I did it by myself." %}

  • {% translate "By + time expressions: For regular intervals" %}

    {% translate "He gets paid by the hour." %}

  • {% translate "By + gerund: To describe methods" %}

    {% translate "You can improve by practicing daily." %}

{% translate "2. The Preposition 'For'" %}

{% translate "Primary Functions" %}

{% translate "Function" %} {% translate "Explanation" %} {% translate "Examples" %}
{% translate "Purpose" %} {% translate "Indicates why something is done" %} {% translate "This tool is for cutting wood." %}
{% translate "Beneficiary" %} {% translate "Shows who benefits from an action" %} {% translate "I bought flowers for my mother." %}
{% translate "Duration" %} {% translate "Expresses how long something lasts" %} {% translate "We waited for three hours." %}
{% translate "Exchange" %} {% translate "Indicates a trade or substitution" %} {% translate "I'll trade you my sandwich for your apple." %}

{% translate "Idiomatic Uses of 'For'" %}

  • {% translate "For + noun: To express purpose" %}

    {% translate "This is for emergencies only." %}

  • {% translate "For + gerund: After certain adjectives" %}

    {% translate "This tool is useful for opening jars." %}

  • {% translate "For + period: With present perfect" %}

    {% translate "I've lived here for five years." %}

{% translate "3. The Preposition 'From'" %}

{% translate "Key Applications" %}

{% translate "Function" %} {% translate "Explanation" %} {% translate "Examples" %}
{% translate "Origin" %} {% translate "Shows where someone/something comes from" %} {% translate "She's from Canada." %}
{% translate "Source" %} {% translate "Indicates where something originates" %} {% translate "I got this information from the internet." %}
{% translate "Time starting point" %} {% translate "Marks when something begins" %} {% translate "The store is open from 9 a.m." %}
{% translate "Material change" %} {% translate "Shows transformation" %} {% translate "Wine is made from grapes." %}

{% translate "Special Patterns with 'From'" %}

  • {% translate "From...to: For ranges" %}

    {% translate "We work from Monday to Friday." %}

  • {% translate "Prevent/protect from: For avoidance" %}

    {% translate "This cream protects from sunburn." %}

  • {% translate "Different from: For comparisons" %}

    {% translate "Your opinion is different from mine." %}
