{% translate "These specialized adverbs serve crucial functions in forming questions, connecting ideas, and establishing relationships between clauses in complex sentences." %}

{% translate "1. Interrogative Adverbs" %}

{% translate "Forming Information Questions" %}

{% translate "Adverb" %} {% translate "Function" %} {% translate "Example Questions" %} {% translate "Possible Answers" %}
where {% translate "Asks about place" %} {% translate "Where do you live?" %} {% translate "In Paris" %}
when {% translate "Asks about time" %} {% translate "When does the train leave?" %} {% translate "At 3:15 PM" %}
why {% translate "Asks about reason" %} {% translate "Why are you crying?" %} {% translate "Because I'm happy" %}
how {% translate "Asks about manner/condition" %} {% translate "How did you solve the problem?" %} {% translate "By thinking creatively" %}

{% translate "Special Interrogative Forms" %}

  • {% translate "How + adjective/adverb: For specific qualities" %}

    {% translate "How tall are you?" %}

    {% translate "How quickly can you type?" %}

    {% translate "How come you're not coming?" %}

{% translate "2. Relative Adverbs" %}

{% translate "Connecting Clauses to Nouns" %}

{% translate "Adverb" %} {% translate "Function" %} {% translate "Example Sentences" %} {% translate "Noun Being Modified" %}
where {% translate "Refers to place" %} {% translate "This is the town where I was born." %} {% translate "town" %}
when {% translate "Refers to time" %} {% translate "Summer is the season when we travel." %} {% translate "season" %}
why {% translate "Refers to reason" %} {% translate "That's the reason why I called you." %} {% translate "reason" %}

{% translate "Alternative Structures" %}

  • {% translate "Preposition + which:" %}

    {% translate "The house where I live." %}

    {% translate "The house in which I live." %}

  • {% translate "That or omission:" %}

    {% translate "The reason why I left." %}

    {% translate "The reason that I left." %}

    {% translate "The reason I left." %}

{% translate "3. Conjunctive Adverbs" %}

{% translate "Linking Independent Clauses" %}

{% translate "Relationship" %} {% translate "Common Adverbs" %} {% translate "Example Sentences" %} {% translate "Punctuation" %}
{% translate "Addition" %} furthermore, moreover, additionally {% translate "She's qualified; furthermore, she's bilingual." %} {% translate "semicolon + comma" %}
{% translate "Contrast" %} however, nevertheless, nonetheless {% translate "It was raining; however, we went hiking." %} {% translate "semicolon + comma" %}
{% translate "Cause/Effect" %} therefore, consequently, thus {% translate "He forgot his passport; therefore, he missed the flight." %} {% translate "semicolon + comma" %}
{% translate "Time" %} meanwhile, subsequently, finally {% translate "She was studying; meanwhile, her friends were partying." %} {% translate "semicolon + comma" %}

{% translate "Flexible Positioning" %}

  • {% translate "Between sentences:" %}

    {% translate "The results were disappointing. However, we learned valuable lessons." %}

  • {% translate "Within sentences:" %}

    {% translate "We were tired; we decided to continue, however." %}

  • {% translate "Parenthetical use:" %}

    {% translate "The project was, nevertheless, a success." %}

{% translate "Common Errors and Special Notes" %}

  • {% translate "Confusing relative adverbs with relative pronouns:" %}

    {% translate "Correct: The day when we met." %}

    {% translate "Incorrect: The day which we met." %}

  • {% translate "Punctuation with conjunctive adverbs:" %}

    {% translate "Correct: It was late; therefore, we left." %}

    {% translate "Incorrect: It was late, therefore we left." %}

  • {% translate "Overusing conjunctive adverbs: Vary your transitional phrases" %}
  • {% translate "How vs. what ... like:" %}

    {% translate "How is your teacher?" %} (asking about health/condition)

    {% translate "What is your teacher like?" %} (asking about personality)
