{% translate "Special Cases and Notes" %}

{% translate "1. Far → Farther/Further Distinction" %}

  • {% translate "Farther/farthest for physical distance" %}

    {% translate "The second store is farther than the first one." %}

  • {% translate "Further/furthest for metaphorical or figurative distance" %}

    {% translate "Let me explain this further." %}

    {% translate "That's the furthest thing from my mind." %}

{% translate "2. Late vs. Lately" %}

  • {% translate "Late (irregular) refers to time" %}

    {% translate "She arrived late, but he arrived later." %}

  • {% translate "Lately (regular) means 'recently'" %}

    {% translate "I've been seeing her more lately." %}

{% translate "Important Note on 'Good' vs 'Well'" %}

{% translate "Remember that good is an adjective while well is typically an adverb (though it can be an adjective when referring to health):" %}

  • {% translate "She is a good singer. (adjective describing 'singer')" %}
  • {% translate "She sings well. (adverb describing 'sings')" %}
  • {% translate "She doesn't feel well. (adjective describing 'she')" %}

{% translate "Common Errors with Irregular Adverbs" %}

  • {% translate "Using adjective forms as adverbs:" %}

    {% translate "She sings good." %}

    {% translate "She sings well." %}

  • {% translate "Creating double comparatives:" %}

    {% translate "He runs more better now." %}

    {% translate "He runs better now." %}

  • {% translate "Confusing 'bad' and 'badly':" %}

    {% translate "He behaved bad." %}

    {% translate "He behaved badly." %}

  • {% translate "Misusing 'less' and 'least':" %}

    {% translate "She visits fewer often now." %}

    {% translate "She visits less often now." %}
