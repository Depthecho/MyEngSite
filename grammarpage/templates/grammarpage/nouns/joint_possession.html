{% translate "Joint Possession" %}

{% translate "Joint possession occurs when two or more nouns own the same thing. The way you use apostrophes and the possessive 's changes depending on whether the ownership is shared or individual." %}

{% translate "Shared Possession" %}

{% translate "When two or more nouns share a single item, ownership is indicated by adding ’s to the last noun in the list only." %}

{% translate "Rule" %} {% translate "Example" %} {% translate "Meaning" %}
{% translate "Possessive is added to the last noun." %} {% translate "Maria and Tom's new car is red." %} {% translate "They own one car together." %}
{% translate "Possessive is added to the last noun." %} {% translate "The students and the teacher's project was a success." %} {% translate "They all worked on one project together." %}

{% translate "Separate Possession" %}

{% translate "When two or more nouns possess their own separate items, ownership is indicated by adding ’s to each noun individually." %}

{% translate "Rule" %} {% translate "Example" %} {% translate "Meaning" %}
{% translate "Possessive is added to each noun." %} {% translate "Maria's and Tom's cars are parked outside." %} {% translate "Maria owns one car, and Tom owns another." %}
{% translate "Possessive is added to each noun." %} {% translate "The student's and the teacher's books are on the table." %} {% translate "The student's book is on the table, and the teacher's book is also on the table." %}
