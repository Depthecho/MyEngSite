{% translate "Compound Nouns" %}

{% translate "A compound noun is a noun made up of two or more words. These words can be a combination of a noun, an adjective, or a verb. The individual words combine to create a new word with a distinct meaning." %}

{% translate "Forms of Compound Nouns" %}

{% translate "Compound nouns can be written in three different ways:" %}

  • {% translate "Closed Form" %}

    {% translate "Written as a single word, with no spaces or hyphens." %}

    {% translate "toothbrush" %}, {% translate "classroom" %}, {% translate "keyboard" %}
  • {% translate "Hyphenated Form" %}

    {% translate "Written with a hyphen between the words." %}

    {% translate "mother-in-law" %}, {% translate "sister-in-law" %}, {% translate "check-in" %}
  • {% translate "Open Form" %}

    {% translate "Written as separate words, but together they function as a single noun." %}

    {% translate "post office" %}, {% translate "bus stop" %}, {% translate "living room" %}
{% translate "Form" %} {% translate "Example" %} {% translate "Sentence" %}
{% translate "Closed" %} {% translate "firefighter" %} {% translate "The firefighter saved the cat." %}
{% translate "Hyphenated" %} {% translate "passer-by" %} {% translate "A passer-by saw the accident." %}
{% translate "Open" %} {% translate "ice cream" %} {% translate "I want a bowl of ice cream." %}

{% translate "Forming Plurals of Compound Nouns" %}

{% translate "The way you form the plural of a compound noun depends on its structure." %}

  • {% translate "Most common nouns:" %}

    {% translate "Add '-s' to the end of the noun." %}

    {% translate "keyboard" %} -> {% translate "keyboards" %}
  • {% translate "Hyphenated nouns:" %}

    {% translate "Add '-s' to the most significant word (usually the noun)." %}

    {% translate "mother-in-law" %} -> {% translate "mothers-in-law" %} {% translate "passer-by" %} -> {% translate "passers-by" %}
  • {% translate "Open form nouns:" %}

    {% translate "Add '-s' to the end of the noun part of the phrase." %}

    {% translate "bus stop" %} -> {% translate "bus stops" %} {% translate "living room" %} -> {% translate "living rooms" %}
